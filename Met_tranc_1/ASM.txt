.386
.MODEL FLAT, STDCALL
EXTRN ExitProcess@4:PROC
.DATA
_ST DD 6 dup(0)
a DD ?
b DD ?
x DD ?
.CODE
MAIN PROC
MOV EAX, 3
MOV a, EAX
MOV EAX, 2
MOV b, EAX
MOV EAX, a
ADD EAX, b
MOV DWORD PTR[_ST + 0], EAX
MOV EAX, DWORD PTR[_ST + 0]
SUB EAX, 5
MOV DWORD PTR[_ST + 4], EAX
MOV EAX, DWORD PTR[_ST + 4]
SUB EAX, 10
MOV DWORD PTR[_ST + 8], EAX
MOV EAX, DWORD PTR[_ST + 8]
ADD EAX, 15
MOV DWORD PTR[_ST + 12], EAX
MOV EAX, DWORD PTR[_ST + 12]
MOV x, EAX
PUSH 0
CALL ExitProcess@4
MAIN ENDP
END MAIN
